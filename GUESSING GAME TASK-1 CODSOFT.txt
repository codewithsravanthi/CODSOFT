import java.util.Random;
import java.util.Scanner;

public class GuessingGame {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();
        int minRange = 1;
        int maxRange = 100;
        int maxAttempts = 5;
        int totalRounds = 0;
        int totalAttempts = 0;
        int totalWins = 0;

        System.out.println("Welcome to the Guessing Game!");

        boolean playAgain = true;
        while (playAgain) {
            int randomNumber = random.nextInt(maxRange - minRange + 1) + minRange;
            int attempts = 0;

            System.out.println("I've selected a random number between 1 and 100. Try to guess it!");

            while (attempts < maxAttempts) {
                attempts++;
                totalAttempts++;

                System.out.print("Attempt " + attempts + ": Enter your guess: ");
                int guess = scanner.nextInt();

                if (guess == randomNumber) {
                    System.out.println("Congratulations! You guessed the correct number in " + attempts + " attempts.");
                    totalWins++;
                    break;
                } else if (guess < randomNumber) {
                    System.out.println("Too low! Try again.");
                } else {
                    System.out.println("Too high! Try again.");
                }
            }

            if (attempts >= maxAttempts) {
                System.out.println("Sorry! You have used all " + maxAttempts + " attempts. The correct number was: " + randomNumber);
            }

            totalRounds++;

            // Calculate and display score
            int scoreBasedOnAttempts = totalAttempts; // Score based on total attempts
            int scoreBasedOnRoundsWon = totalWins; // Score based on total rounds won

            System.out.println("Score based on attempts: " + scoreBasedOnAttempts);
            System.out.println("Score based on rounds won: " + scoreBasedOnRoundsWon);

            System.out.print("Do you want to play again? (yes/no): ");
            String playAgainInput = scanner.next();
            playAgain = playAgainInput.equalsIgnoreCase("yes");
        }

        System.out.println("Game over!");
        System.out.println("Total rounds played: " + totalRounds);
        System.out.println("Total wins: " + totalWins);
        System.out.println("Total attempts: " + totalAttempts);

        scanner.close();
    }
}
